def is_palendrome(s):
    """Check whether a string is a palindrome"""
    for i in range(len(s) - 1, -1, -1):
        if s[i + 1] == s[i]:
            return False
    return True

def is_even(i):
    """Check whether an integer is even"""
    return i % 2
